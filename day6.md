# DNSとは
━━━━━━━━━━━━━━━━━━━

DNSとは：

ドメイン名を
IPアドレスへ変換する仕組み

━━━━━━━━━━━━━━━━━━━
# ドメインとは
━━━━━━━━━━━━━━━━━━━

人間が理解しやすい：

文字の名前。

例：

google.com
youtube.com

━━━━━━━━━━━━━━━━━━━
# IPアドレスとは
━━━━━━━━━━━━━━━━━━━

コンピュータ同士が通信するための：

「住所」

例：

172.217.221.139

━━━━━━━━━━━━━━━━━━━
# 名前解決とは
━━━━━━━━━━━━━━━━━━━

名前解決とは：

ドメイン名
↓
IPアドレス

へ変換すること。

例：

google.com
↓
172.217.xxx.xxx

━━━━━━━━━━━━━━━━━━━

━━━━━━━━━━━━━━━━━━━

① DNSへ問い合わせ
↓
② IP取得
↓
③ そのIPへ通信
