# HTTPとは
━━━━━━━━━━━━━━━━━━━

HTTPとは：

「暗号化されていない通信」

━━━━━━━━━━━━━━━━━━━
# HTTPの危険性
━━━━━━━━━━━━━━━━━━━

HTTPでは：

・password
・session_id
・cookie

などが：

通信途中で見える危険がある。

━━━━━━━━━━━━━━━━━━━
# HTTPSとは
━━━━━━━━━━━━━━━━━━━

HTTPSとは：

「暗号化されたHTTP通信」

━━━━━━━━━━━━━━━━━━━
# HTTPSのS
━━━━━━━━━━━━━━━━━━━

HTTPS の：

S

は：

Secure

を意味する。

━━━━━━━━━━━━━━━━━━━
# HTTPSの本質
━━━━━━━━━━━━━━━━━━━

HTTPSは：

① 通信を暗号化する
② サイトが本物か確認する

ための仕組み。

━━━━━━━━━━━━━━━━━━━
# 暗号化とは
━━━━━━━━━━━━━━━━━━━

通常HTTP：

password=1234

━━━━━━━━━━━━━━━━━━━

HTTPS：

k29dk2kdk29x...

━━━━━━━━━━━━━━━━━━━

のように、

通信内容を読めない形に変換する。

━━━━━━━━━━━━━━━━━━━
# 鍵マーク
━━━━━━━━━━━━━━━━━━━

ブラウザURL左側の：

🔒

マークは：

「HTTPS通信中」

を意味する。

━━━━━━━━━━━━━━━━━━━
# 鍵マークの意味
━━━━━━━━━━━━━━━━━━━

・暗号化通信中
・証明書あり
・HTTPS有効

を表す。

━━━━━━━━━━━━━━━━━━━

を証明するもの。

━━━━━━━━━━━━━━━━━━━

CNとは：

「どのサイト用証明書か」

を表す。

例：

example.com

━━━━━━━━━━━━━━━━━━━

発行者とは：

「誰が本物だと保証したか」

を表す。

━━━━━━━━━━━━━━━━━━━
# 認証局（CA）
━━━━━━━━━━━━━━━━━━━

証明書を発行する会社を：

CA
(Certificate Authority)

という。

今回確認した：

Cloudflare

もその一種。

━━━━━━━━━━━━━━━━━━━
# 有効期限
━━━━━━━━━━━━━━━━━━━

証明書には期限がある。

期限切れになると：

・危険表示
・ブラウザ警告
・HTTPSエラー

になる。

━━━━━━━━━━━━━━━━━━━
# 公開鍵
━━━━━━━━━━━━━━━━━━━

公開鍵とは：

「暗号化通信で使用する鍵情報」

━━━━━━━━━━━━━━━━━━━
# HTTPS通信
━━━━━━━━━━━━━━━━━━━

HTTPSでは：

・公開鍵
・秘密鍵

を使い安全通信を行う。

━━━━━━━━━━━━━━━━━━━
# なぜHTTPSが必要か
━━━━━━━━━━━━━━━━━━━

ログイン時：

・password
・session_id

などを安全に守るため。

━━━━━━━━━━━━━━━━━━━
# session_idの危険性
━━━━━━━━━━━━━━━━━━━

session_id を盗まれると：

「ログイン済みユーザー」

になりすませる可能性がある。

━━━━━━━━━━━━━━━━━━━
# Cookieとの関係
━━━━━━━━━━━━━━━━━━━

Cookieには：

session_id

が保存される。

例：

session_id=abc123

━━━━━━━━━━━━━━━━━━━
# Secure属性
━━━━━━━━━━━━━━━━━━━

Secureとは：

「HTTPS時だけCookie送信」

する設定。

━━━━━━━━━━━━━━━━━━━
# Secureの本質
━━━━━━━━━━━━━━━━━━━

「危険なHTTP通信ではCookie送らない」

ための安全設定。

━━━━━━━━━━━━━━━━━━━
# Secureは暗号化ではない
━━━━━━━━━━━━━━━━━━━

暗号化しているのは：

HTTPS（TLS/SSL）

━━━━━━━━━━━━━━━━━━━

Secureは：

「HTTPS時だけCookie送れ」

という追加ルール。

━━━━━━━━━━━━━━━━━━━
# HTTPSとSecureの違い
━━━━━━━━━━━━━━━━━━━

HTTPS
↓
通信全体を暗号化

━━━━━━━━━━━━━━━━━━━

Secure
↓
HTTPS時だけCookie送信

━━━━━━━━━━━━━━━━━━━
# HTTPとHTTPSの違い
━━━━━━━━━━━━━━━━━━━

HTTP
↓
暗号化なし
危険

━━━━━━━━━━━━━━━━━━━

HTTPS
↓
暗号化あり
安全

━━━━━━━━━━━━━━━━━━━

━━━━━━━━━━━━━━━━━━━

HTTPSは：

・通信を暗号化する
・サイトが本物か確認する
・session_idを守る
・passwordを守る

ために存在する。
